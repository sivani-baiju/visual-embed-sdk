Configuration to define the customization on the Embedded Thoughtspot components.

example
 init({
   // ...
   customizations: {
    style: {
      customCSS: {
        variables: {},
        rules_UNSTABLE: {}
      }
    },
    content: {
     strings: {
       'LIVEBOARDS': 'Dashboards'
     }
    },
    iconSpriteUrl: 'https://my-custom-icon-sprite.svg'
   }
 })
